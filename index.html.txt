    <script>
        class PDFTextExtractor {
            constructor() {
                this.currentPdf = null;
                this.extractedText = '';
                this.isDarkMode = false;
                this.searchTerm = '';
                
                this.initializeElements();
                this.setupEventListeners();
                this.setupPDFJS();
                this.loadTheme();
            }

            initializeElements() {
                this.elements = {
                    pdfInput: document.getElementById('pdfInput'),
                    extractBtn: document.getElementById('extractBtn'),
                    searchInput: document.getElementById('searchInput'),
                    searchBtn: document.getElementById('searchBtn'),
                    clearSearchBtn: document.getElementById('clearSearchBtn'),
                    downloadBtn: document.getElementById('downloadBtn'),
                    themeToggle: document.getElementById('themeToggle'),
                    progress: document.getElementById('progress'),
                    progressFill: document.getElementById('progressFill'),
                    statusText: document.getElementById('statusText'),
                    progressText: document.getElementById('progressText'),
                    textOutput: document.getElementById('textOutput'),
                    fileInfo: document.getElementById('fileInfo'),
                    fileName: document.getElementById('fileName'),
                    fileSize: document.getElementById('fileSize'),
                    stats: document.getElementById('stats'),
                    pageCount: document.getElementById('pageCount'),
                    charCount: document.getElementById('charCount'),
                    wordCount: document.getElementById('wordCount'),
                    ocrOption: document.getElementById('ocrOption'),
                    enableOCR: document.getElementById('enableOCR')
                };
            }

            setupPDFJS() {
                // Configure PDF.js worker
                pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/4.6.82/pdf.worker.min.js';
            }

            setupEventListeners() {
                this.elements.pdfInput.addEventListener('change', (e) => this.handleFileSelect(e));
                this.elements.extractBtn.addEventListener('click', () => this.extractText());
                this.elements.searchBtn.addEventListener('click', () => this.searchText());
                this.elements.clearSearchBtn.addEventListener('click', () => this.clearSearch());
                this.elements.downloadBtn.addEventListener('click', () => this.downloadText());
                this.elements.themeToggle.addEventListener('click', () => this.toggleTheme());
                this.elements.searchInput.addEventListener('input', () => this.handleSearchInput());
                this.elements.searchInput.addEventListener('keypress', (e) => {
                    if (e.key === 'Enter') this.searchText();
                });
            }

            handleFileSelect(event) {
                const file = event.target.files[0];
                if (!file) return;

                if (file.type !== 'application/pdf') {
                    alert('Please select a PDF file.');
                    return;
                }

                this.currentPdf = file;
                this.elements.fileName.textContent = file.name;
                this.elements.fileSize.textContent = this.formatFileSize(file.size);
                this.elements.fileInfo.classList.remove('hidden');
                this.elements.extractBtn.disabled = false;
                this.elements.ocrOption.classList.remove('hidden');
                
                // Reset previous results
                this.resetOutput();
            }

            formatFileSize(bytes) {
                if (bytes === 0) return '0 Bytes';
                const k = 1024;
                const sizes = ['Bytes', 'KB', 'MB', 'GB'];
                const i = Math.floor(Math.log(bytes) / Math.log(k));
                return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
            }

            resetOutput() {
                this.elements.textOutput.textContent = '';
                this.elements.textOutput.classList.add('hidden');
                this.elements.stats.classList.add('hidden');
                this.elements.downloadBtn.disabled = true;
                this.elements.searchBtn.disabled = true;
                this.elements.searchInput.value = '';
                this.clearSearch();
            }

            async extractText() {
                if (!this.currentPdf) return;

                this.showProgress();
                this.elements.extractBtn.disabled = true;

                try {
                    const arrayBuffer = await this.readFileAsArrayBuffer(this.currentPdf);
                    const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
                    
                    let fullText = '';
                    const totalPages = pdf.numPages;
                    
                    for (let pageNum = 1; pageNum <= totalPages; pageNum++) {
                        this.updateProgress(pageNum, totalPages, `Extracting page ${pageNum} of ${totalPages}...`);
                        
                        const page = await pdf.getPage(pageNum);
                        const textContent = await page.getTextContent();
                        
                        // Check if page has text content
                        const hasText = textContent.items.length > 0;
                        
                        if (hasText) {
                            // Extract text normally
                            const pageText = textContent.items.map(item => item.str).join(' ');
                            fullText += `\n\n--- Page ${pageNum} ---\n\n${pageText}`;
                        } else if (this.elements.enableOCR.checked) {
                            // Perform OCR for scanned pages
                            this.updateProgress(pageNum, totalPages, `Performing OCR on page ${pageNum}...`);
                            const ocrText = await this.performOCR(page);
                            fullText += `\n\n--- Page ${pageNum} (OCR) ---\n\n${ocrText}`;
                        } else {
                            fullText += `\n\n--- Page ${pageNum} ---\n\n[No text content found - this may be a scanned image. Enable OCR for better results.]`;
                        }
                    }

                    this.extractedText = fullText.trim();
                    this.displayText(this.extractedText);
                    this.updateStats(totalPages);
                    this.elements.downloadBtn.disabled = false;
                    this.elements.searchBtn.disabled = false;
                    
                } catch (error) {
                    console.error('Error extracting text:', error);
                    alert('Error extracting text from PDF: ' + error.message);
                } finally {
                    this.hideProgress();
                    this.elements.extractBtn.disabled = false;
                }
            }

            async performOCR(page) {
                try {
                    // Get page as image for OCR
                    const viewport = page.getViewport({ scale: 2.0 });
                    const canvas = document.createElement('canvas');
                    const context = canvas.getContext('2d');
                    
                    canvas.width = viewport.width;
                    canvas.height = viewport.height;
                    
                    await page.render({
                        canvasContext: context,
                        viewport: viewport
                    }).promise;
                    
                    return new Promise((resolve, reject) => {
                        Tesseract.recognize(canvas, 'eng', {
                            logger: (m) => {
                                if (m.status === 'recognizing text') {
                                    this.updateProgress(m.progress * 100, 100, `OCR: ${Math.round(m.progress * 100)}%`);
                                }
                            }
                        }).then(({ data: { text } }) => {
                            resolve(text);
                        }).catch(reject);
                    });
                } catch (error) {
                    return `[OCR failed: ${error.message}]`;
                }
            }

            readFileAsArrayBuffer(file) {
                return new Promise((resolve, reject) => {
                    const reader = new FileReader();
                    reader.onload = (e) => resolve(e.target.result);
                    reader.onerror = reject;
                    reader.readAsArrayBuffer(file);
                });
            }

            displayText(text) {
                this.elements.textOutput.textContent = text;
                this.elements.textOutput.classList.remove('hidden');
            }

            updateStats(totalPages) {
                const charCount = this.extractedText.length;
                const wordCount = this.extractedText.split(/\s+/).filter(word => word.length > 0).length;
                
                this.elements.pageCount.textContent = totalPages;
                this.elements.charCount.textContent = charCount.toLocaleString();
                this.elements.wordCount.textContent = wordCount.toLocaleString();
                this.elements.stats.classList.remove('hidden');
            }

            searchText() {
                const searchTerm = this.elements.searchInput.value.trim();
                if (!searchTerm || !this.extractedText) return;

                this.searchTerm = searchTerm;
                this.highlightSearchResults();
                this.elements.clearSearchBtn.classList.remove('hidden');
            }

            handleSearchInput() {
                const searchTerm = this.elements.searchInput.value.trim();
                if (searchTerm === '') {
                    this.clearSearch();
                }
            }

            highlightSearchResults() {
                const text = this.extractedText;
                const regex = new RegExp(`(${this.escapeRegExp(this.searchTerm)})`, 'gi');
                const highlightedText = text.replace(regex, '<span class="highlight">$1</span>');
                
                this.elements.textOutput.innerHTML = highlightedText;
                
                // Scroll to first occurrence
                const firstHighlight = this.elements.textOutput.querySelector('.highlight');
                if (firstHighlight) {
                    firstHighlight.scrollIntoView({ behavior: 'smooth', block: 'center' });
                }
            }

            clearSearch() {
                this.searchTerm = '';
                this.elements.searchInput.value = '';
                this.elements.clearSearchBtn.classList.add('hidden');
                
                if (this.extractedText) {
                    this.displayText(this.extractedText);
                }
            }

            escapeRegExp(string) {
                return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
            }

            downloadText() {
                if (!this.extractedText) return;

                const blob = new Blob([this.extractedText], { type: 'text/plain' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `extracted_text_${new Date().getTime()}.txt`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
            }

            toggleTheme() {
                this.isDarkMode = !this.isDarkMode;
                document.documentElement.setAttribute('data-theme', this.isDarkMode ? 'dark' : 'light');
                this.elements.themeToggle.textContent = this.isDarkMode ? '‚òÄÔ∏è' : 'üåô';
                localStorage.setItem('darkMode', this.isDarkMode);
            }

            loadTheme() {
                const savedTheme = localStorage.getItem('darkMode');
                if (savedTheme === 'true') {
                    this.isDarkMode = true;
                    document.documentElement.setAttribute('data-theme', 'dark');
                    this.elements.themeToggle.textContent = '‚òÄÔ∏è';
                }
            }

            showProgress() {
                this.elements.progress.classList.remove('hidden');
                this.updateProgress(0, 100, 'Starting extraction...');
            }

            hideProgress() {
                this.elements.progress.classList.add('hidden');
            }

            updateProgress(current, total, status) {
                const percentage = Math.round((current / total) * 100);
                this.elements.progressFill.style.width = `${percentage}%`;
                this.elements.progressText.textContent = `${percentage}%`;
                this.elements.statusText.textContent = status;
            }
        }

        // Initialize the application when DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            new PDFTextExtractor();
        });
    </script>
